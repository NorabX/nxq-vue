## Getting Started

Install nxq-vue via npm: `npm install nxq-vue`

```
// index.html

<div id="app">
    <div id="div1">Hello, World</div>
</div>
<script src="app.js"></script>

// app.js
import nxq from './node_modules/nxq-vue/nxq.js'

const app = new Vue({
    el: '#app',
    mixins:[nxq],
    mounted() {
        console.log(this.q('#div1').qText())
    }
});

// output
Hello, World
```
